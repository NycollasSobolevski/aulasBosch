######## comandos: #######


mkdir nycollas 				// cria um diretorio 'nycollas'

cd nycollas 				// entra na pasta
cd ..                // Nivel acima (pasta anterior)


git version 				// versao do git

git init 				// diz para o git que aquela pasta e o repositorio, e ele vai criar uma 						pasta oculta dentro da pasta principal("Nycollas").


git status 				// mostra como ta o repositorio

git add arquivo.txt 			// o item 'arquivo.txt' passa a ser gerenciado pelo git

git add .				// adiciona todos os itens que nao foram adicionados

git commit -m "primeiro commit"		// faz um comit do arquivo (-m é mensagem para vvoce lembrar o que foi 						feito) ele funciona como um CheckPoint


git config --global user.name "Your name" //global para ele usar o mesmo nick para a maquina toda, e local somente 	  						para aquele repositorio 
dit config --global user.email "email"



					//se eu alterar o documento, em seguida dar um 'git status' ele vai mostrar 						que o arquivo foi alterado e nao comitado

					//sempre que for commitar em txt quando for incluir uma linha nova, tem que 						adicionar o arquivo novamente 

git commit -a				// Abre um TXT para voce nomear o commit 


git log 				// mostra o historico de alterações
    git log --oneline  //  tudo em uma linha
